Usage in Components

Components with shadow DOM can access these variables:

// In your Lit component
static styles = css\`
  :host {
    font-family: var(--font-family-base);
    color: var(--text-color);
    background: var(--bg-color);
  }

  .header {
    background: var(--header-bg);
    color: var(--header-text);
  }

  .button-primary {
    background: var(--button-bg);
    color: var(--button-text);
  }

  .button-primary:hover {
    background: var(--button-hover);
  }

  /* Use font-scale for responsive sizing */
  .content {
    font-size: calc(1rem * var(--font-scale));
  }
\`;

JavaScript API

// Import the theme manager
import { themeManager } from '/js/theme-manager.js';

// Set theme
themeManager.setTheme('dark');
themeManager.toggleTheme();

// Get current theme
const theme = themeManager.getCurrentTheme(); // 'light' or 'dark'

// Set custom variable
themeManager.setVariable('my-custom-color', '#ff0000');

// Get variable
const color = themeManager.getVariable('color-primary');

// Font scaling
themeManager.setFontScale(1.2);
themeManager.updateResponsiveFontScale(containerWidth);

// Observe theme changes
themeManager.addObserver((themeName) => {
    console.log('Theme changed to:', themeName);
});
